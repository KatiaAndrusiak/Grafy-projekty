# Zadanie 1
    python zad1.py --gnp {n} {p}  przyklad python zad1.py --gnp 5 0.3	   
    n [int] - liczba wierzcholkow
    p [float] - prawdopodobienstwo wygenerowania krawedzi pomiedzy dwoma wierzcholkami
    --help - flaga pomocnicza
